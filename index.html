alpha = 0.01

Gradient Descent

In [5]:
for i in range(iterations):
    p = theta[0] - alpha/m * sum(h - y)
    q = theta[1] - alpha/m * sum((h - y) * x)
    theta[0] = p
    theta[1] = q
    h = theta[0] + theta[1] * x

plt.plot(x, y, 'rx')
